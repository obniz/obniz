Optional keys

keys: undefined | string

Overrides SmpEncryptOptions.keys

  • Defined in src/obniz/libs/embeds/bleHci/bleRemotePeripheral.ts:96

Use pairing keys

// Javascript Example

const keys = "xxxxx";
await obniz.ble.initWait({});
obniz.ble.scan.onfind = function(peripheral){
if(peripheral.localName == "my peripheral"){
     peripheral.onconnect = async function(){
         console.log("success");
         await peripheral.pairingWait({keys});  // pairing with stored keys.

     }
     await peripheral.connectWait();
    }
}
await obniz.ble.scan.startWait();

passkeyCallback: undefined | (() => Promise<number>)

Overrides SmpEncryptOptions.passkeyCallback

  • Defined in src/obniz/libs/embeds/bleHci/bleRemotePeripheral.ts:125

Callback function that call on pairing passkey required.

// Javascript Example

const keys = "xxxxx";
await obniz.ble.initWait({});
obniz.ble.scan.onfind = function(peripheral){
if(peripheral.localName == "my peripheral"){
     peripheral.onconnect = async function(){
         console.log("success");
         let passkeyCallback = await ()=>{
             let number = prompt("Please type passkey code."); //HTML prompt
             return number;
         }
         await peripheral.pairingWait({passkeyCallback});  // pairing with user input passkey.

     }
     await peripheral.connectWait();
    }
}
await obniz.ble.scan.startWait();
